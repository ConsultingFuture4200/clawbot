Use `set -uo pipefail` (NOT `set -e` — we want to run all checks even if some fail). Use a counter pattern:

```bash
PASS=0
FAIL=0
TOTAL=0

check() {
  local name="$1"
  local result="$2"  # 0 = pass, non-zero = fail
  TOTAL=$((TOTAL + 1))
  if [ "$result" -eq 0 ]; then
    PASS=$((PASS + 1))
    echo "  PASS: $name"
  else
    FAIL=$((FAIL + 1))
    echo "  FAIL: $name"
  fi
}
```

**GMAIL-01: OAuth authentication for both accounts**
```bash
echo "=== GMAIL-01: OAuth Authentication ==="

# Check gog has 2+ authenticated accounts
ACCOUNT_COUNT=$(gog auth list --json 2>/dev/null | jq 'length' 2>/dev/null || echo 0)
[ "$ACCOUNT_COUNT" -ge 2 ]
check "gog has >= 2 authenticated accounts (found: $ACCOUNT_COUNT)" $?

# Check personal account can access Gmail API
PERSONAL_EMAIL=$(jq -r '.email' /sandbox/state/email-sync-personal.json 2>/dev/null)
gog gmail messages search "in:inbox" --max 1 --account "$PERSONAL_EMAIL" > /dev/null 2>&1
check "Personal account ($PERSONAL_EMAIL) Gmail API accessible" $?

# Check work account can access Gmail API
WORK_EMAIL=$(jq -r '.email' /sandbox/state/email-sync-work.json 2>/dev/null)
gog gmail messages search "in:inbox" --max 1 --account "$WORK_EMAIL" > /dev/null 2>&1
check "Work account ($WORK_EMAIL) Gmail API accessible" $?
```

**GMAIL-03: history.list incremental sync**
```bash
echo ""
echo "=== GMAIL-03: Incremental Sync (history.list) ==="

# Verify history.list works for personal account
PERSONAL_HID=$(jq -r '.historyId' /sandbox/state/email-sync-personal.json 2>/dev/null)
[ "$PERSONAL_HID" != "null" ] && [ -n "$PERSONAL_HID" ]
check "Personal account has valid historyId ($PERSONAL_HID)" $?

gog gmail history --account "$PERSONAL_EMAIL" --since "$PERSONAL_HID" > /dev/null 2>&1
check "Personal account history.list returns successfully" $?

# Verify history.list works for work account
WORK_HID=$(jq -r '.historyId' /sandbox/state/email-sync-work.json 2>/dev/null)
[ "$WORK_HID" != "null" ] && [ -n "$WORK_HID" ]
check "Work account has valid historyId ($WORK_HID)" $?

gog gmail history --account "$WORK_EMAIL" --since "$WORK_HID" > /dev/null 2>&1
check "Work account history.list returns successfully" $?

# Verify state files are persisted to disk (not memory-only)
[ -f /sandbox/state/email-sync-personal.json ]
check "Personal sync state file persisted at /sandbox/state/email-sync-personal.json" $?

[ -f /sandbox/state/email-sync-work.json ]
check "Work sync state file persisted at /sandbox/state/email-sync-work.json" $?
```
